-- lefthook.yml --
min_version: 1.1.1
pre-commit:
  stash_unstaged_changes: false
  commands:
    edit_file:
      run: |
        if test -n "$RUN_STAGE_FIXED_TEST"; then
          echo newline >> "[file].js"
          echo newline >> file.txt
        fi
      stage_fixed: true
